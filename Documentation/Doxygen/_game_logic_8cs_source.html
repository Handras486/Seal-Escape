121  public void ApproachAll()
122  {
123  var toRemoveEnemies = new List<IEnemyModel>();
124  var toRemoveFood = new List<IFoodModel>();
125 
126  // Move enemies
127  foreach (var enemy in this.EnemyModelLogicPairs)
128  {
129  if (enemy.Key.HasAimBot)
130  {
131  enemy.Value.AimBotApproach(this.PlayerModelLogicPair.Key.YPosition);
132  }
133  else
134  {
135  enemy.Value.HorizontalApproach();
136  }
137 
138  // Collision check with player
139  if (this.PlayerModelLogicPair.Key.IsCollision(enemy.Key))
140  {
141  this.PlayerModelLogicPair.Value.DecreaseLivesLeft(enemy.Key.Damage);
142  toRemoveEnemies.Add(enemy.Key);
143  if (this.PlayerModelLogicPair.Key.LivesLeft <= 0)
144  {
145  this.GameOver?.Invoke(this, EventArgs.Empty);
146  }
147  }
148  }
149 
150  // Move food
151  foreach (var food in this.FoodModelLogicPairs)
152  {
153  if (food.Key.HasRandomApproach)
154  {
155  food.Value.RandomApproach();
156  }
157  else
158  {
159  food.Value.HorizontalApproach();
160  }
161 
162  // Collision check with player
163  if (this.PlayerModelLogicPair.Key.IsCollision(food.Key))
164  {
165  int multiply = 0;
166  foreach (var item in this.GameModel.ActivePowerUps)
167  {
168  multiply += item.MultipliesCollectedFoodBy;
169  }
170 
171  if (multiply == 0)
172  {
173  multiply = 1;
174  }
175 
176  this.IncreaseFoodCollected(food.Key.Value * multiply);
177  toRemoveFood.Add(food.Key);
178  }
179  }
180 
181  // Run cleanup
182  foreach (var item in toRemoveEnemies)
183  {
184  this.GameModel.EnemiesOnScreen.Remove(item);
185  this.EnemyModelLogicPairs.Remove(item);
186  }
187 
188  foreach (var item in toRemoveFood)
189  {
190  this.GameModel.FoodOnScreen.Remove(item);
191  this.FoodModelLogicPairs.Remove(item);
192  }
193 
194  // Add to score
195  this.scoreCounter++;
196  if (this.scoreCounter % 5 == 0)
197  {
198  this.GameModel.Score++;
199  }
200 
201  // Check and set level
202  if (this.GameModel.Score < 200)
203  {
204  this.GameModel.Level = 1;
205  }
206  else
207  {
208  this.GameModel.Level = 2;
209  }
210  }

266  public void GenerateMap()
267  {
268  // Enemies
269  int enemiesToAddCount = 10 - this.GameModel.EnemiesOnScreen.Count;
270  for (int i = 0; i < enemiesToAddCount; i++)
271  {
272  double xpos = (double)this.rng.Next(
273  (int)this.GameModel.GameWidth,
274  (int)this.GameModel.GameWidth * 2);
275  double ypos = (double)this.rng.Next(150, (int)this.GameModel.GameHeight - 500);
276  int dmg;
277  bool aimbot;
278  if (this.rng.Next(0, 10) < 8)
279  {

282  }

286  aimbot = true;

291  bool addIsOk = true;

293  {
294  if (enemyToAdd.IsCollision(item) ||
295  (Math.Abs(enemyToAdd.YPosition - item.YPosition) < 100 &&
296  Math.Abs(enemyToAdd.XPosition - item.XPosition) < 200) ||
297  (item.Damage == 3 && enemyToAdd.Damage == 3))
298  {
299  addIsOk = false;
300  }
301  }
302 
303  foreach (var item in this.GameModel.FoodOnScreen)
304  {
305  if (enemyToAdd.IsCollision(item))
306  {
307  addIsOk = false;
308  }
309  }
310 
311  if (addIsOk)
312  {
313  this.GameModel.EnemiesOnScreen.Add(enemyToAdd);
314  }
315  }

317  // Food
